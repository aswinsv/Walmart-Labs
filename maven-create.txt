
#To create the maven project
mvn archetype:generate -DgroupId=com.walmartlabs -DartifactId=ticket-reservation -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false

mvn package

java -cp target/ticket-reservation-1.0-SNAPSHOT.jar com.walmartlabs.App
